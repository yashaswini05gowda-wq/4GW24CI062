#include <stdio.h>
#include <stdlib.h>
/
/ Employee structure
typedef struct Employee
{
char SSN[20];
char Name[50];
char Dept[30];
char Designation[30];
float Sal;
char PhNo[15];
struct Employee *prev, *next;
} Employee;
Employee *head = NULL;
/
/ Function to create a new node using the Structure Employee as data type to //return function.
Employee *createNode()
{
Employee *newNode = (Employee *)malloc(sizeof(Employee));
if (!newNode)
{
printf("Memory allocation failed!\n");
exit(1);
}
printf("Enter SSN, Name, Dept, Designation, Salary, Phone:\n");
scanf("%s %s %s %s %f %s",
newNode->SSN, newNode->Name, newNode->Dept,
newNode->Designation, &newNode->Sal, newNode->PhNo);
newNode->prev = newNode->next = NULL;
return newNode;
}void insertEnd(Employee *newNode)
{
if (head == NULL)
{
head = newNode;
return;
}
Employee *temp = head;
while (temp->next != NULL)
temp = temp->next;
temp->next = newNode;
newNode->prev = temp;
}
/
/ Insert at front
void insertFront(Employee *newNode)
{
if (head == NULL)
{
head = newNode;
return;
}
newNode->next = head;
head->prev = newNode;
head = newNode;
}
/
/ Delete at end
void deleteEnd()
{
if (head == NULL)
{
printf("List is empty!\n");
return;
}
Employee *temp = head;
if (temp->next == NULL)
{
free(temp);
head = NULL;
return;
}while (temp->next != NULL)
temp = temp->next;
temp->prev->next = NULL;
free(temp);
}
/
/ Delete at front
void deleteFront()
{
if (head == NULL)
{
printf("List is empty!\n");
return;
}
Employee *temp = head;
if (temp->next == NULL)
{
free(temp);
head = NULL;
return;
}
head = head->next;
head->prev = NULL;
free(temp);
}
/
/ Display DLL and count nodes
void display()
{
if (head == NULL)
{
printf("List is empty!\n");
return;
}
Employee *temp = head;
int count = 0;
printf("\nDLL Status:\n");
while (temp != NULL)
{
printf("SSN: %s, Name: %s, Dept: %s, Designation: %s, Salary: %.2f, Phone: %s\n",
temp->SSN, temp->Name, temp->Dept, temp->Designation, temp->Sal, temp->PhNo);
temp = temp->next;
count++;
}
printf("Total Employees: %d\n", count);
}
/
/ Demonstrate DLL as Double Ended Queue using user input
void dequeDemo()
{
printf("\n--- Demonstrating DLL as Double Ended Queue ---\n");
/
/ Insert at rear
printf("\nInsert at rear (End):\n");
insertEnd(createNode());
display();
/
/ Insert at front
printf("\nInsert at front:\n");
insertFront(createNode());
display();
/
/ Delete from rear
printf("\nDelete from rear (End):\n");
deleteEnd();
display();
/
/ Delete from front
printf("\nDelete from front:\n");
deleteFront();
display();
printf("\n--- End of Deque Demonstration ---\n");
}
/
/ Menu
int main()
{
int choice, n, i;
while (1)
{
printf("\n--- MENU ---\n");

printf("6. Delete at Front\n");

scanf("%d", &choice);
switch (choice)
{
case 1:
printf("Enter number of employees: ");

for (i = 0; i < n; i++)
{
insertEnd(createNode());
}
break;
case 2:
display();
break;
case 3:
insertEnd(createNode());
break;
case 4:
deleteEnd();
break;

insertFront(createNode());
break;

deleteFront();

case 7:

}
