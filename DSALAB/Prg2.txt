#include <stdio.h>
#include <ctype.h>
#include <math.h>
#define MAX 100
/
/ Stack implementation
int stack[MAX];
int top = -1;
void push(int val)
{
if (top == MAX - 1)
{
printf("Stack Overflow\n");
return;
}
stack[++top] = val;
}
int pop()
{
if (top == -1)
{
printf("Stack Underflow\n");
return -1;
}
return stack[top--];
}
